# Simos Tuning Suite — Python dependencies
#
# Full install (includes hardware deps):
#   pip install -r requirements.txt
#   pip install git+https://github.com/bri3d/sa2_seed_key.git
#
# CI / headless-only install (no BLE/CAN):
#   pip install numpy
#   pip install git+https://github.com/bri3d/sa2_seed_key.git
#
# Windows EXE build — use build.bat, which installs everything automatically.
#
# J2534 note: PassThru DLL is loaded at runtime via ctypes — no pip package.
# Install your J2534 DLL separately (Tactrix, Mongoose, VNCI, etc.)

# ── Core protocol ─────────────────────────────────────────────────────────────
udsoncan >= 1.21              # UDS client — all flash/read/routine operations

# ── Transport ─────────────────────────────────────────────────────────────────
bleak >= 0.21                 # BLE transport for ESP32 bridge
pyserial >= 3.5               # USB ISO-TP bridge (CP210x / CH340)
python-can >= 4.0             # SocketCAN backend (Linux)

# ── Calibration math ──────────────────────────────────────────────────────────
numpy >= 1.24                 # CAL table parsing, axis math

# ── Crypto ────────────────────────────────────────────────────────────────────
pycryptodome >= 3.18          # AES-CBC for Simos12/18 flash blocks

# ── SA2 seed/key ──────────────────────────────────────────────────────────────
# Install from source — not on PyPI:
#   pip install git+https://github.com/bri3d/sa2_seed_key.git
# sa2_seed_key

# ── Build tooling (not needed at runtime) ─────────────────────────────────────
